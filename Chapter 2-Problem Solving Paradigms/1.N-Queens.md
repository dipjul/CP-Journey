# N-Queens
```java
class Nqueens {
    int[] row;
    boolean[] rw, ld, rd;
    Nqueens(int n) {
        row = new int[n];
        rw = new boolean[n];
        ld = new boolean[2*n-1];
        rd = new boolean[2*n-1];
        List<List<String>> ans = new ArrayList<>();
        backtrack(0, n, ans);
        return ans;
    }
    private void backtrack(int c, int n, List<List<String>> ans) {
        if(c == n) {
          // row array will have the ans
          ans.add(printer(row, n));
        }
        for(int r = 0; r < n; r++) {
            if(!rw[r] && !ld[r - c + n - 1] && !rd[r + c]) {
                rw[r] = ld[r - c + n - 1] = rd[r + c] = true;
                row[c] = r;
                backtrack(c+1,n,ans);
                rw[r] = ld[r - c + n - 1] = rd[r + c] = false;
            }
        }
    }
    
    // Helper function
    private List<String> printer(int[] row, int n) {
        List<String> res = new ArrayList<>();
        for(int i = 0; i < n; i++) {
            StringBuilder sb = new StringBuilder(n);
            for(int j = 0; j < n; j++) {
                if(j == row[i])
                    sb.append('Q');
                else
                    sb.append('.');
            }
            res.add(sb.toString());
        }
        return res;
    }
}
```
